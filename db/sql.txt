-- Dumping structure for procedure diklatpim_e_dupak.xget_data_pegawai_per_pegawai
DELIMITER //
CREATE DEFINER=`root`@`localhost` PROCEDURE `xget_data_pegawai_per_pegawai`(
  IN `prm_nip` VARCHAR(50)

)
BEGIN
  SELECT a.*, b.nama_skpd, b.nip_pimpinan, b.nama_pimpinan, b.jabatan_pimpinan, c.nama_jabatan, d.alias AS kategori, e.jenjang, f.golru2 as golongan, f.pangkat 
  FROM tb_pegawai a 
  LEFT JOIN ref_skpd b ON a.id_skpd = b.id
  LEFT JOIN ref_jabfung c ON a.id_jabatan = c.id
  LEFT JOIN ref_jabfung_kategori d ON a.id_kategori = d.id
  LEFT JOIN ref_jabfung_jenjang e ON a.id_jenjang = e.id
  LEFT JOIN ref_golru f ON a.id_golru = f.id_golru
  WHERE a.nip = prm_nip;
END//
DELIMITER ;

-- Dumping structure for procedure diklatpim_e_dupak.xtampil_log_book_per_pegawai
DELIMITER //
CREATE DEFINER=`root`@`localhost` PROCEDURE `xtampil_log_book_per_pegawai`(
  IN `prm_IDjab` INT,
  IN `prm_IDkat` INT,
  IN `prm_IDjen` INT,
  IN `prm_nip` VARCHAR(50),
  IN `prm_bln` INT,
  IN `prm_thn` INT


)
BEGIN
  SELECT a.*, 
       c.unsur,
         b.urutan, b.butir_kegiatan, b.satuan_hasil, b.angka_kredit,
       a.jumlah_logbook*b.angka_kredit AS jumlah_angka_kredit
  FROM tb_log_book a 
  LEFT JOIN ref_butir_kegiatan b ON a.id_unsur = b.id_unsur AND a.id_urutan = b.urutan
  LEFT JOIN ref_butir_kegiatan_unsur c ON a.id_unsur = c.id_unsur
  WHERE a.id_jabatan  = prm_IDjab 
  AND   a.id_kategori = prm_IDkat
  AND   a.id_jenjang  = prm_IDjen
  AND   a.nip  = prm_nip
  AND   MONTH(a.tgl_logbook) = prm_bln
  AND   YEAR(a.tgl_logbook) = prm_thn
  GROUP BY id_unsur, id_urutan;
END//
DELIMITER ;
